function enviar() { const btn = document.querySelector('button'); btn.disabled = true; btn.innerText = "PROCESANDO..."; const datos = { action: "generar_codigo", nombre: document.getElementById('nombre').value, dni: document.getElementById('dni').value, area: document.getElementById('area').value, turno: document.getElementById('turno').value, dieta: document.getElementById('dieta').value }; fetch(API, { method: 'POST', mode: 'cors', // Agregado para evitar el bloqueo body: JSON.stringify(datos) }) .then(res => res.json()) .then(res => { if(res.status === "success") { document.getElementById('codigo').innerText = res.codigo; document.getElementById('resultado').style.display = 'block'; btn.innerText = "GENERAR OTRO"; btn.disabled = false; } else { alert("Error: " + res.message); btn.disabled = false; btn.innerText = "REINTENTAR"; } }) .catch(err => { console.error(err); alert("Hubo un problema de conexión con el servidor."); btn.disabled = false; btn.innerText = "REINTENTAR"; }); }
