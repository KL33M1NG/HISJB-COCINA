// ========== EDITAR MENÚ ========== if (action === "editar_menu" && e.parameter.platoPrincipal) { var sheetMenu = ss.getSheetByName("Menu_Dia"); var data = sheetMenu.getDataRange().getValues(); var hoy = new Date(); var hoyStr = Utilities.formatDate(hoy, ss.getSpreadsheetTimeZone(), "dd/MM/yyyy"); var turnoEditar = e.parameter.turno.toUpperCase(); // Buscar la fila del menú de hoy para ese turno for (var i = 1; i < data.length; i++) { if (data[i][0] && typeof data[i][0] === 'object') { var fechaStr = Utilities.formatDate(new Date(data[i][0]), ss.getSpreadsheetTimeZone(), "dd/MM/yyyy"); if (fechaStr === hoyStr && data[i][1].toString().toUpperCase() === turnoEditar && data[i][5] === "PUBLICADO") { // Actualizar la fila existente var dietas = { GENERAL: e.parameter.general || e.parameter.platoPrincipal, HIPOSÓDICA: e.parameter.hiposodica || e.parameter.platoPrincipal, DIABÉTICA: e.parameter.diabetica || e.parameter.platoPrincipal, CELIACA: e.parameter.celiaca || e.parameter.platoPrincipal, GÁSTRICA: e.parameter.gastrica || e.parameter.platoPrincipal }; sheetMenu.getRange(i + 1, 3).setValue(e.parameter.platoPrincipal.toUpperCase()); sheetMenu.getRange(i + 1, 4).setValue(e.parameter.postre ? e.parameter.postre.toUpperCase() : ""); sheetMenu.getRange(i + 1, 5).setValue(JSON.stringify(dietas)); return darSalidaJSONP(callback, { status: "success", message: "Menú actualizado correctamente" }); } } } return darSalidaJSONP(callback, { status: "error", message: "No se encontró el menú para editar" }); }
